## Fingerprint Caching

### Project Fingerprint

```python
def _project_fingerprint(artifacts: list[Artifact]) -> str:
    # Exclude audit output types from fingerprint (they change every run)
    source_fps = sorted(
        art.fingerprint or compute_fingerprint(art.body)
        for art in artifacts
        if art.type not in {"challenge", "audit"}
    )
    return hashlib.sha256("|".join(source_fps).encode()).hexdigest()[:16]
```

### Cache Hit/Miss Logic

```python
def _apply_fingerprint_cache(artifacts: list[Artifact], cache_dir: Path) -> tuple[bool, list]:
    proj_fp = _project_fingerprint(artifacts)
    cached = _load_cached_findings(cache_dir, proj_fp)
    if cached:
        return True, cached
    return False, []
```

**Cache Storage**: `.specflow/audits/.cache/<fingerprint>.md`
- YAML frontmatter with `fingerprint`, `cached_at`, `findings`
- Markdown body with full report

**Benefit**: Re-audit of unchanged project completes in <1 second.

## Analysis Axes

### Horizontal Analysis

Per artifact type, check internal consistency:

```python
def _horizontal_analysis(artifacts: list[Artifact]) -> dict[str, list[dict]]:
    findings = {}
    for type_name in {"requirement", "architecture", "detailed-design", 
                      "story", "decision", "audit", "challenge"}:
        type_arts = [a for a in artifacts if a.type == type_name]
        if not type_arts:
            continue
        
        # Check: all in draft?
        if all(a.status == "draft" for a in type_arts):
            findings.setdefault(type_name, []).append({
                "severity": "warn",
                "message": f"All {len(type_arts)} {type_name} artifacts are in draft"
            })
        
        # Check: no tags?
        if type_name == "challenge" and not any(a.tags for a in type_arts):
            findings.setdefault(type_name, []).append({
                "severity": "info",
                "message": f"No challenge artifacts use tags"
            })
    return findings
```

### Vertical Analysis

Per top-level REQ, check V-model thread coherence:

```python
def _vertical_analysis(artifacts: list[Artifact]) -> list[dict]:
    findings = []
    reqs = [a for a in artifacts if get_prefix_from_id(a.id) == "REQ"]
    
    for req in reqs:
        # Find ARCHs refining this REQ
        archs = [a for a in artifacts 
                if any(l.target == req.id and l.role == "refined_by" for l in a.links)]
        if not archs:
            findings.append({
                "severity": "warn",
                "message": f"{req.id}: no ARCH refinement in V-model thread"
            })
        
        # Find STORYs implementing this REQ
        stories = [a for a in artifacts 
                  if any(l.target == req.id and l.role == "implements" for l in a.links)]
        for story in stories:
            # Check test verification
            tests = [a for a in artifacts 
                    if any(l.target == story.id and l.role == "verified_by" for l in a.links)]
            if not tests:
                findings.append({
                    "severity": "warn",
                    "message": f"{story.id} (implements {req.id}): no test verification"
                })
    return findings
```

### Cross-Cutting Analysis

Project-wide concerns:

```python
def _cross_cutting_analysis(artifacts: list[Artifact], root: Path) -> dict[str, list[dict]]:
    findings = {}
    
    # Baseline drift
    baselines = list_baselines(root)
    if len(baselines) >= 2:
        diff = diff_baselines(baselines[-2], baselines[-1])
        if diff["modified"]:
            findings.setdefault("baseline_drift", []).append({
                "severity": "info",
                "message": f"{len(diff['modified'])} artifacts changed since last baseline"
            })
    
    # Standards compliance
    installed = list_installed_standards(root)
    if installed:
        for std in installed:
            gaps = gap_analysis(artifacts, [std])
            if gaps[std.name]["uncovered"] > 0:
                findings.setdefault("compliance", []).append({
                    "severity": "warn",
                    "message": f"{gaps[std.name]['uncovered']} uncovered clauses in {std.name}"
                })
    
    return findings
```

## Challenge Deduplication

```python
def _create_chl_artifacts(root: Path, findings: list[dict], target_id: str) -> int:
    # Load existing challenges to avoid duplicates
    existing = discover_artifacts(root, artifact_type="challenge")
    seen_keys = {f"{chl.title}:{','.join(sorted(l.target for l in chl.links))}" 
                 for chl in existing}
    
    count = 0
    for f in findings:
        if f["severity"] == "info":
            continue
        title = f["message"][:100]
        key = f"{title}:{target_id}"
        if key in seen_keys:
            continue
        seen_keys.add(key)
        
        # Create challenge artifact
        result = create_artifact(root, artifact_type="challenge", ...)
        update_artifact(root, result["id"], severity=f["severity"], technique="project-audit")
        count += 1
    return count
```

**Key Insight**: Deduplication by `(title, link_target)` means re-running audit on unchanged warnings produces zero new CHLs.

## Acceptance Criteria

1. Cache hit reuses previous findings (no re-analysis)
2. Vertical analysis flags missing ARCH refinements and missing test verifications
3. Cross-cutting analysis detects baseline drift and compliance gaps
4. Challenge deduplication prevents duplicate CHLs on re-runs
5. Audit completes in <5 seconds for 300 artifacts (with cache)
